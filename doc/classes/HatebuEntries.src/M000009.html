    def marge_summary_entries(entries)
        des=Array.new
        while entry=entries.pop
            tag_str=String.new
            while tag = entry[:tag].pop
                tag_str=tag_str + "["+tag+"]"
            end
            if entry[:summary]
                entry[:summary]=tag_str+entry[:summary]
            else
                entry[:summary]=tag_str
            end
                des.push entry
        end
        return des
    end
