    def get_entry(id)
        path="/atom/edit/#{id}"
        res = @http.get(path,
                   'Accept' => 'application/x.atom+xml, application/xml, text/xml, */*',
                   'X-WSSE'=> @credential_string )
        #puts res.body
        parse_entry(res.body)

    end
